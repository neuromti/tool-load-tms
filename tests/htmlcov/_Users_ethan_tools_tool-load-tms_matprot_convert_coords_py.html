86def get_M1(hemisphere="L", cosys="MNI") -> Coordinate: 

87 """ 

88 According to Mayka, M. A., Corcos, D. M., Leurgans, S. E., & Vaillancourt, D. E. (2006): 

89 Three-dimensional locations and boundaries of motor and premotor cortices as defined by functional brain imaging: a meta-analysis. NeuroImage, 31(4), 1453–1474. https://doi.org/10.1016/j.neuroimage.2006.02.004 

90 Consider that any MNI coordinates not reported in Talairach space were converted using the transformation equations for above the AC line (z > 0): 

91 xV = 0.9900x yV = 0.9688y + 0.0460z, zV = -0.0485y + 0.9189z 

92 xyz_in_Tailarach = [-37, -21, 58]; 

93 Tailarach2MNI = [0.9900,0,0;0,0.9688,0.0460;0,-0.0485,0.9189]; 

94 xyz_in_MNI = (Tailarach2MNI*xyz_in_Tailarach')'; 

95 Result as literals for speed 

96 

97 See also 

98 https://neuroimage.usc.edu/brainstorm/CoordinateSystems 

99 https://www.brainvoyager.com/bv/doc/UsersGuide/CoordsAndTransforms/CoordinateSystems.html 

100 

101 """ 

102 if cosys == "MNI": 

103 M1 = [-36.6300, -17.6768, 54.3147] 

104 elif cosys == "Tailarach": 

105 M1 = [-37.0, -21.0, 58.0] 

106 else: 

107 raise NotImplementedError("Unknown coordinate system") 

108 if hemisphere == "R": 

109 M1[0] = -M1[0] 

110 return M1 

111 elif hemisphere == "L": 

112 return M1 

113 else: 

114 raise NotImplementedError("Unknown hemisphere")

117def classify_hemisphere( 

118 coords: List[Coordinate], 

119) -> Tuple[List[str], Coordinate, Coordinate]: 

120 """for each coordinate, assign whether it lies in the right or left hemisphere 

121 

122 args 

123 ---- 

124 coords: List[Coordinate] 

125 a list of target coordinates 

126 

127 returns 

128 ------- 

129 hemisphere: List[str] 

130 a list of strings (either "L", or "R", or "V" for vertex line) describing the hemisphere 

131 lCoG: Coordinate 

132 Center of Gravity of coordinates assigned to left hemisphere 

133 rCoG: Coordinate 

134 Center of Gravity of coordinates assigned to right hemisphere 

135 

136 """ 

137 

138 hemi = [] 

139 right = [] 

140 left = [] 

141 for pos in coords: 

142 if pos[0] < 0: 

143 left.append(pos) 

144 hemi.append("L") 

145 elif pos[0] > 0: 

146 right.append(pos) 

147 hemi.append("R") 

148 else: 

149 hemi.append("V") 

150 

151 MISSING_COG = [np.NaN] * 3 

152 lCoG = np.mean(left, 0).tolist() if len(left) != 0 else MISSING_COG 

153 rCoG = np.mean(right, 0).tolist() if len(right) != 0 else MISSING_COG 

154 return hemi, lCoG, rCoG

157def shift_origin(coords: List[List[float]]) -> List[List[float]]: 

158 hemi, lcog, rcog = classify_hemisphere(coords) 

159 rshift = np.atleast_2d(rcog) - np.atleast_2d(get_M1("R")) 

160 lshift = np.atleast_2d(lcog) - np.atleast_2d(get_M1("L")) 

161 vshift = np.atleast_2d([0, rshift[0][1], rshift[0][2]]) 

162 shifted = [] 

163 for h, pos in zip(hemi, coords): 

164 if h == "R": 

165 shifted.append((pos - rshift).tolist()[0]) 

166 elif h == "L": 

167 shifted.append((pos - lshift).tolist()[0]) 

168 else: 

169 shifted.append((pos - vshift).tolist()[0]) 

170 

171 return shifted
